USAGE EXAMPLES
================================================================================

Creating a Simple Command:
  export const myCommand: Command = {
    data: new SlashCommandBuilder()
      .setName('mycmd')
      .setDescription('Does something'),
    
    async execute(interaction) {
      await interaction.reply(
        ResponseBuilder.success('Done')
          .addField('Result', 'Success!')
          .build()
      );
    }
  };

With Cooldown & Permissions:
  export const adminCommand: Command = {
    data: new SlashCommandBuilder()...,
    cooldown: 5000,
    requiredPermissions: [PermissionFlagsBits.Administrator],
    
    async execute(interaction) {
      // CommandManager handles permissions & cooldown
      // Your command logic here
    }
  };

With Full Validation:
  async execute(interaction) {
    if (!await ValidationHelpers.requireGuild(interaction)) return;
    
    const user = interaction.options.getUser('user', true);
    const member = await ValidationHelpers.requireMember(interaction, user.id);
    if (!member) return;
    
    await interaction.reply(ResponseBuilder.success('OK').build());
  }
